Server-Side Integration

Set up a TCP/UDP listener or use TrackPro‚Äôs API to process GPS data.

Node.js TCP Server

Create a TCP server to receive data from GPS trackers:

const net = require('net');

const server = net.createServer((socket) => {
  console.log('Device connected:', socket.remoteAddress);

  socket.on('data', (data) => {
    console.log('Received:', data.toString());
    const message = data.toString();
    if (message.startsWith('+RESP')) {
      const [header, imei, lat, lon, speed, ...rest] = message.split(',');
      console.log(`IMEI: ${imei}, Lat: ${lat}, Lon: ${lon}, Speed: ${speed}`);
    }
    socket.write('ACK');
  });

  socket.on('end', () => {
    console.log('Device disconnected');
  });

  socket.on('error', (err) => {
    console.error('Socket error:', err);
  });
});

server.listen(5000, '0.0.0.0', () => {
  console.log('TrackPro TCP server running on port 5000');
});
  1. Install: npm install net
  2. Run: node server.js
  3. Point devices to trackpro.arybit.co.ke:5000
  4. Parse protocols per manufacturer manuals

TrackPro API

Store and process data via TrackPro‚Äôs RESTful API:

const axios = require('axios');

async function sendToTrackPro(imei, latitude, longitude, speed, timestamp) {
  try {
    const response = await axios.post('https://api.trackpro.arybit.co.ke/v1/tracking', {
      imei,
      latitude,
      longitude,
      speed,
      timestamp
    }, {
      headers: { 'Authorization': 'Bearer your_api_key' }
    });
    console.log('Data sent to TrackPro:', response.data);
  } catch (error) {
    console.error('Error:', error.message);
  }
}
